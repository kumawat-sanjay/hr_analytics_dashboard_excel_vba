Private Sub btnAdd_Click()

    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Clean_Data")
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    
    ws.Cells(lastRow, GetColumnIndex(ws, "EmployeeID")).Value = txtEmpID.Value
    ws.Cells(lastRow, GetColumnIndex(ws, "EmployeeName")).Value = txtName.Value
    ws.Cells(lastRow, GetColumnIndex(ws, "Department")).Value = txtDept.Value
    ws.Cells(lastRow, GetColumnIndex(ws, "Salary")).Value = txtSalary.Value
    ws.Cells(lastRow, GetColumnIndex(ws, "TotalExperience")).Value = txtExp.Value
    
    MsgBox "Employee Added ?"

End Sub

Private Sub btnSearch_Click()

    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Clean_Data")
    
    Dim colID As Long
    colID = GetColumnIndex(ws, "EmployeeID")
    
    Dim resultRow As Variant
    resultRow = Application.Match(txtEmpID.Value, ws.Columns(colID), 0)
    
    If IsError(resultRow) Then
        MsgBox "Employee Not Found"
        Exit Sub
    End If
    
    txtName.Value = ws.Cells(resultRow, GetColumnIndex(ws, "EmployeeName")).Value
    txtDept.Value = ws.Cells(resultRow, GetColumnIndex(ws, "Department")).Value
    txtSalary.Value = ws.Cells(resultRow, GetColumnIndex(ws, "Salary")).Value
    txtExp.Value = ws.Cells(resultRow, GetColumnIndex(ws, "TotalExperience")).Value

End Sub
